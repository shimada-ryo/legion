# Phase 3 a04: Web UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Phase 3 で server に出た `blackboardMessages` + `blackboard.message` WS event を Web UI に表示する。Phase 2 narrow の 4 タブ (Canvas / Events / Overview / Diff) に **Blackboard タブ** を加え、Reviewer の decision を Overview で表示、EventLogPane に Blackboard 重畳表示モードを足す。Reviewer ロールノードは既存 StatusNode に乗る (機構的には自然に追従)。

**Architecture:** `InstanceDetail` ページが `agentInstances` に加えて `blackboardMessages` を fetch。`SidebarTabs` に Blackboard 列を加え、`BlackboardTab` コンポーネントが表示。WS subscriber は `blackboard.message` event 受信時に local state を更新。OverviewTab は agentInstance の `roleName === 'reviewer'` のとき、blackboard から `system.review.decision` topic の最新 payload を引いて表示。

**Tech Stack:** React、Vite、Vitest、@testing-library/react、xyflow (Canvas)。既存 Phase 2 narrow UI 構造をそのまま継承。

**Spec reference:** [docs/dev/specs/2026-05-14_phase3_design.md](../specs/2026-05-14_phase3_design.md) § 8.1〜8.3。

**Depends on:** [a03 server expansion plan](2026-05-14_phase3_a03_server.md) (API + WS event 経路)。

```typescript
// packages/web/src/components/sidebar-tabs/Blackboard.tsx
import { useState } from 'react'
import type { BlackboardMessageView } from '../../types/blackboard'
import type { AgentInstanceView } from '../../types/agent-instance'

interface BlackboardTabProps {
  blackboardMessages: BlackboardMessageView[]
  agentInstances: AgentInstanceView[]
}

type FilterMode = 'all' | 'system' | 'user'

export function BlackboardTab({ blackboardMessages, agentInstances }: BlackboardTabProps) {
  const [filter, setFilter] = useState<FilterMode>('all')
  const [expandedId, setExpandedId] = useState<string | null>(null)

  const filtered = blackboardMessages.filter((m) => {
    if (filter === 'system') return m.topic.startsWith('system.')
    if (filter === 'user') return !m.topic.startsWith('system.')
    return true
  })

  if (filtered.length === 0 && blackboardMessages.length === 0) {
    return <div className="empty">No blackboard messages yet.</div>
  }

  return (
    <div className="blackboard-tab">
      <div className="filters">
        <button
          type="button"
          aria-pressed={filter === 'all'}
          onClick={() => setFilter('all')}
        >
          All
        </button>
        <button
          type="button"
          aria-pressed={filter === 'system'}
          onClick={() => setFilter('system')}
        >
          system.*
        </button>
        <button
          type="button"
          aria-pressed={filter === 'user'}
          onClick={() => setFilter('user')}
        >
          user
        </button>
      </div>
      <ul className="messages">
        {filtered.map((m) => {
          const agent = m.publisherAgentId
            ? agentInstances.find((a) => a.id === m.publisherAgentId)
            : null
          return (
            <li
              key={m.id}
              data-topic={m.topic}
              onClick={() => setExpandedId(expandedId === m.id ? null : m.id)}
            >
              <span className="time">{new Date(m.publishedAt).toISOString().slice(11, 19)}</span>
              <span className="topic">{m.topic}</span>
              {agent && <span className="publisher">{agent.roleNodeId}</span>}
              {expandedId === m.id && (
                <pre className="payload">{JSON.stringify(m.payload, null, 2)}</pre>
              )}
            </li>
          )
        })}
      </ul>
    </div>
  )
}
```

```typescript
interface OverviewProps {
  selected: AgentInstanceView | null
  agentInstances: AgentInstanceView[]
  blackboardMessages: BlackboardMessageView[]   // ★ Phase 3
  template: WorkflowTemplate
}

export function OverviewTab({ selected, agentInstances, blackboardMessages, template }: OverviewProps) {
  if (!selected) return <div>Select a node to see details.</div>

  const children = agentInstances.filter((a) => a.parentAgentInstanceId === selected.id)

  // Reviewer の decision を blackboard から引く
  const decisionMsg = selected.roleNodeId === 'reviewer'
    ? [...blackboardMessages].reverse().find((m) =>
        m.topic === 'system.review.decision' &&
        (m.payload as { agentInstanceId?: string })?.agentInstanceId === selected.id,
      )
    : undefined
  const decision = (decisionMsg?.payload as { decision?: string; feedback?: string } | undefined)

  return (
    <div className="overview">
      <dl>
        <dt>Role</dt><dd>{selected.roleNodeId}</dd>
        <dt>Status</dt><dd>{selected.status}</dd>
        {/* ... existing fields */}
      </dl>

      {decision && (
        <section className="decision">
          <h4>Decision</h4>
          <p data-decision={decision.decision}>{decision.decision}</p>
          {decision.feedback && <p className="feedback">{decision.feedback}</p>}
        </section>
      )}

      {children.length > 0 && (
        <section className="children">
          <h4>Spawned</h4>
          <ul>
            {children.map((c) => (
              <li key={c.id} data-id={c.id}>{c.id} ({c.roleNodeId})</li>
            ))}
          </ul>
        </section>
      )}
    </div>
  )
}
```

```typescript
interface EventLogPaneProps {
  events: AgentEvent[]
  agentInstances: AgentInstanceView[]
  blackboardMessages: BlackboardMessageView[]   // ★ Phase 3
}

export function EventLogPane({ events, agentInstances, blackboardMessages }: EventLogPaneProps) {
  const [agentFilter, setAgentFilter] = useState<string | null>(null)
  const [showBlackboard, setShowBlackboard] = useState(false)   // ★ Phase 3

  // existing: events を agentFilter で絞る
  const filteredEvents = agentFilter
    ? events.filter((e) => /* ... existing logic */)
    : events

  // Phase 3: Blackboard message を時系列に挿入
  const blackboardItems: Array<{ kind: 'bb'; m: BlackboardMessageView }> = showBlackboard
    ? blackboardMessages.map((m) => ({ kind: 'bb', m }))
    : []
  const eventItems = filteredEvents.map((e) => ({ kind: 'ev' as const, e }))
  const merged = [...eventItems, ...blackboardItems].sort((a, b) => {
    const ta = a.kind === 'ev' ? a.e.timestamp.getTime() : a.m.publishedAt
    const tb = b.kind === 'ev' ? b.e.timestamp.getTime() : b.m.publishedAt
    return ta - tb
  })

  return (
    <div>
      <div className="filters">
        {/* existing agent chips */}
        <button
          type="button"
          aria-pressed={showBlackboard}
          onClick={() => setShowBlackboard((v) => !v)}
        >
          + Blackboard
        </button>
      </div>
      <ul>
        {merged.map((item, i) => item.kind === 'ev' ? (
          <li key={i} className="event">{/* existing render */}</li>
        ) : (
          <li key={i} className="blackboard-event">
            <span className="topic">[bb] {item.m.topic}</span>
            <span className="payload">{JSON.stringify(item.m.payload).slice(0, 80)}</span>
          </li>
        ))}
      </ul>
    </div>
  )
}
```
